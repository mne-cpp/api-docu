143  bool result = true;

146 
147  neval = 0;

149 
150  if (report_func != NULL && report > 0) {

152  }

159  ilo = i;
160  if (y[i] > y[ihi]) {

163  } else if (y[i] > y[inhi])

166  }

168  /*

170  */
171  if (count == report && report_func != NULL) {

173  qCritical("Interation interrupted.");

175  break;
176  }

178  }
179  if (rtol < ftol) break;
180  if (neval >= max_eval) {

182  result = false;
183  break;
184  }

201  }
