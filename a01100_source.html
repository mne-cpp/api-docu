134 static char *next_word(FILE *in)
135 
136 {
137  char *next = MALLOC_6(MAXWORD,char);
138  int c;
139  int p,k;
140  int inquote;
141 
142  skip_comments(in);
143 
144  inquote = FALSE;
145  for (k = 0, p = 0, c = fgetc(in); c != EOF && !whitespace_quote(c,inquote) ; c = fgetc(in), k++) {
146  if (k == 0 && c == '"')
147  inquote = TRUE;
148  else
149  next[p++] = c;

152  FREE_6(next);

157  if (c != EOF) {

159  ;
160  if (c != EOF)
161  ungetc(c,in);
162  }
163 #ifdef DEBUG
164  if (next)
165  printf("<%s>\n",next);
166 #endif
